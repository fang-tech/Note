## 递归

```cpp
class Solution {
public:
    vector<int> memo;

    int dfs(int n) {
        if (n < 0) return 0;
        if (n == 0) return 1;
        if (memo[n] != -1) return memo[n];
        memo[n] = dfs(n-1) + dfs(n-2);
        return memo[n];
    }

    int climbStairs(int n) {
        memo.resize(n + 1, -1);
        return dfs(n);
    }
};
```

## 递推

```cpp
class Solution {
public:
    vector<int> f;

    int climbStairs(int n) {
        f.resize(n + 2);
        f[0] = 0, f[1] = 1;
        for (int i = 0; i < n; i++) {
            f[i+2] = f[i] + f[i+1];
        }
        return f[n+1];
    }
};
```
